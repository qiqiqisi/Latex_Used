\subsection{可复用流程图（TikZ 节点）}
下面的流程图不依赖外部图片，可直接修改节点文字、增删节点或调整箭头。

\begin{figure}[H]
  \centering
  \begin{tikzpicture}[
    node distance=8mm,
    >=Latex,
    font=\small,
    process/.style={
      rectangle,rounded corners=3pt,draw=oucblue,fill=ouclight,
      minimum width=6.2cm,minimum height=0.82cm,align=center
    },
    decision/.style={
      diamond,aspect=2.4,draw=oucblue,fill=ouclight,
      minimum width=4.0cm,minimum height=0.9cm,align=center
    },
    terminal/.style={
      process,fill=oucblue,text=white,rounded corners=12pt,
      minimum width=3.3cm
    },
    flow/.style={->,line width=1pt,color=oucblue}
  ]
    \node[terminal] (start) {开始};
    \node[process,below=of start] (input) {读取输入并完成初始化};
    \node[decision,below=of input,yshift=-1mm] (check) {条件是否满足？};
    \node[process,below left=11mm and -4mm of check] (adjust) {调整参数};
    \node[process,below right=11mm and -4mm of check] (run) {执行核心步骤};
    \node[terminal,below=24mm of check] (finish) {输出结果};

    \draw[flow] (start) -- (input);
    \draw[flow] (input) -- (check);
    \draw[flow] (check) -| node[pos=0.25,above]{否} (adjust);
    \draw[flow] (adjust) |- (input);
    \draw[flow] (check) -| node[pos=0.25,above]{是} (run);
    \draw[flow] (run) |- (finish);
  \end{tikzpicture}
  \caption{通用实验流程图模板}
  \label{fig:flowchart}
\end{figure}
